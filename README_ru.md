		}).then(this.handleResponse);

		return this.get('/product').then((data: ApiListResponse<IProduct>) =>
			data.items.map((item) => ({
				...item,

			}))
		);
	}

	/**

	createOrder(order: IOrder): Promise<IOrderResult> {

	}
